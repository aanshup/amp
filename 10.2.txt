----- COMPARE TWO STRINGS IF EQUAL AND NOT EQUAL -----

DATA SEGMENT
S1 DB "HEY"
S2 DB "HELLO"
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START:
MOV AX,DATA
MOV DS,AX
MOV ES,AX

LEA SI,S1
LEA DI,S2
MOV CX,000Dh

COMPARE: MOV AL,[SI]
MOV BL,[DI]
CMP BL,0
JE sequal
CMP AL,BL
JNE nequal

INC SI
INC DI
JMP COMPARE

nequal: MOV DX,0
JMP eop

sequal: MOV DX,0FFFFH
eop: INT 3h

CODE ENDS
END START
