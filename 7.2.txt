----- find even number from array of data bytes -----

DATA SEGMENT
A1 db 84h,23h,0AFh,4Eh,82h,0F4h,75h,99h,33h,10h
eve dw 0015h
X db 100 DUP(0)
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START:
MOV AX,DATA
MOV DS,AX
MOV SI,0000h
MOV CX,000Ah
MOV DI,[eve]
MOV BL,2

BACK:
MOV AL,[SI]
MOV AH,0
DIV BL
CMP AH,000h
JNZ SKIP

MOV AL,[SI]
MOV [DI],AL
INC DI
SKIP: INC SI
LOOP BACK

END START
CODE ENDS
