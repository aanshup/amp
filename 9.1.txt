----- FACTORIAL OF GIVEN NUMBER -----

DATA SEGMENT
A db 0100H DUP(00h)
n db 05h
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START:
MOV AX,DATA
MOV DS,AX

MOV SI,0100H
MOV DI,0101H
MOV AX,0001H

MOV BX,[SI]
FACT:
CMP BX,00H

JE DONE
MUL BX
DEC BX
JMP FACT

DONE:
MOV [DI],AX 

CODE ENDS
END START
