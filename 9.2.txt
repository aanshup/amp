----- FIND SQUARE OF GIVEN 8 BIT NUMBER -----

DATA SEGMENT
A db 0100H DUP(00h)
n db 16
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START:
MOV AX,DATA
MOV DS,AX

MOV SI,0100H
MOV DI,0101H
MOV AX,[SI]
MOV BX,01H
MOV DX,01H

SR:
SUB AX,BX
CMP AX,00H
JE DONE
ADD BX,02H
ADD DX,01H
JMP SR

DONE:
MOV [DI],DX

CODE ENDS
END START
